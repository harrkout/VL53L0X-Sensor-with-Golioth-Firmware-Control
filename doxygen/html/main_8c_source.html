55static int data_received(struct golioth_blockwise_download_ctx *ctx, const uint8_t *data, size_t offset, size_t len, bool last)
56{
57 struct dfu_ctx *dfu = CONTAINER_OF(ctx, struct dfu_ctx, fw_ctx);
58 int err;
59
60 LOG_DBG("Received %zu bytes at offset %zu%s", len, offset, last ? " (last)" : "");
61
62 if (offset == 0) {
63 err = flash_img_prepare(&dfu->flash);
64 if (err) {
65 return err;
66 }
67 }
68
69 err = flash_img_buffered_write(&dfu->flash, data, len, last);
70 if (err) {
71 LOG_ERR("Failed to write to flash: %d", err);
72 return err;
73 }
74

76 err = golioth_fw_report_state(client, "main", current_version_str, dfu->version, GOLIOTH_FW_STATE_DOWNLOADED, GOLIOTH_DFU_RESULT_INITIAL);

79 }

82 if (err) {
83 LOG_ERR("Failed to update to '%s' state: %d", "updating", err);
84 }
85
86 LOG_INF("Requesting upgrade");
87
88 err = boot_request_upgrade(BOOT_UPGRADE_TEST);
89 if (err) {

91 return err;

96 /* Synchronize logs */
97 LOG_PANIC();

130static int golioth_desired_update(const struct coap_packet *update, struct coap_reply *reply, const struct sockaddr *from)
131{
132 struct dfu_ctx *dfu = &update_ctx;
133 struct coap_reply *fw_reply;
134 const uint8_t *payload;
135 uint16_t payload_len;
136 size_t version_len = sizeof(dfu->version) - 1;
137 uint8_t uri[64];
138 uint8_t *uri_p;
139 size_t uri_len = sizeof(uri);
140 int err;
141
142 payload = coap_packet_get_payload(update, &payload_len);
143 if (!payload) {
144 LOG_ERR("No payload in CoAP!");
145 return -EIO;
146 }
147
148 LOG_HEXDUMP_DBG(payload, payload_len, "Desired");
149
150 err = golioth_fw_desired_parse(payload, payload_len, dfu->version, &version_len, uri, &uri_len);
151 if (err) {
152 LOG_ERR("Failed to parse desired version: %d", err);
153 return err;
154 }
155
156 dfu->version[version_len] = '\0';
157
158 if (version_len == strlen(current_version_str) && !strncmp(current_version_str, dfu->version, version_len)) {
159 LOG_INF("Desired version (%s) matches current firmware version!", log_strdup(current_version_str));
160 return -EALREADY;
161 }
162
163 fw_reply = coap_reply_next_unused(coap_replies, ARRAY_SIZE(coap_replies));
164 if (!reply) {
165 LOG_ERR("No more reply handlers");
166 return -ENOMEM;
167 }
168
169 uri_p = uri_strip_leading_slash(uri, &uri_len);
170
171 err = golioth_fw_report_state(client, "main", current_version_str, dfu->version, GOLIOTH_FW_STATE_DOWNLOADING, GOLIOTH_DFU_RESULT_INITIAL);
172 if (err) {
173 LOG_ERR("Failed to update to '%s' state: %d", "downloading", err);
174 }
175
176 err = golioth_fw_download(client, &dfu->fw_ctx, uri_p, uri_len, fw_reply, data_received);
177 if (err) {
178 LOG_ERR("Failed to request firmware: %d", err);
179 return err;
180 }
181
182 return 0;
183}
