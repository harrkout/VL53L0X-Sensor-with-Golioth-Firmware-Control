42static int current_version_init(const struct device *dev)
43{
44 struct mcuboot_img_header hdr;
45 int written;
46 int err;
47
48 err = boot_read_bank_header(FLASH_AREA_IMAGE_PRIMARY, &hdr, sizeof(hdr));
49 if (err) {
50 LOG_ERR("Failed to read primary area (%u) header: %d",
51 FLASH_AREA_IMAGE_PRIMARY, err);
52 return err;
53 }
54
55 written = snprintf(current_version_str, sizeof(current_version_str),
56 "%u.%u.%u",
57 (unsigned int)hdr.h.v1.sem_ver.major,
58 (unsigned int)hdr.h.v1.sem_ver.minor,
59 (unsigned int)hdr.h.v1.sem_ver.revision);
60 if (written >= sizeof(current_version_str)) {
61 LOG_ERR("Version string is too long!");
62 current_version_str[0] = '\0';
63 return -ENOMEM;
64 }
65
66 return 0;
67}

83static int flash_area_check_empty(const struct flash_area *fa,
84 bool *out_empty)
85{
86 uint32_t data[16];
87 off_t addr;
88 off_t end;
89 int bytes_to_read;
90 int rc;
91 int i;
92 uint8_t erased_val;
93 uint32_t erased_val_32;
94
95 __ASSERT_NO_MSG(fa->fa_size % 4 == 0);
96
97 erased_val = flash_area_erased_val(fa);
98 erased_val_32 = ERASED_VAL_32(erased_val);
99
100 end = fa->fa_size;
101 for (addr = 0; addr < end; addr += sizeof(data)) {
102 if (end - addr < sizeof(data)) {
103 bytes_to_read = end - addr;
104 } else {
105 bytes_to_read = sizeof(data);
106 }
107
108 rc = flash_area_read(fa, addr, data, bytes_to_read);
109 if (rc != 0) {
110 flash_area_close(fa);
111 return rc;
112 }
113
114 for (i = 0; i < bytes_to_read / 4; i++) {
115 if (data[i] != erased_val_32) {
116 *out_empty = false;
117 flash_area_close(fa);
118 return 0;
119 }
120 }
121 }
122
123 *out_empty = true;
124
125 return 0;
126}

134static int flash_img_erase_if_needed(struct flash_img_context *ctx)
135{
136 bool empty;
137 int err;
138
139 if (IS_ENABLED(CONFIG_IMG_ERASE_PROGRESSIVELY)) {
140 return 0;
141 }
142
143 err = flash_area_check_empty(ctx->flash_area, &empty);
144 if (err) {
145 return err;
146 }
147
148 if (empty) {
149 return 0;
150 }
151
152 err = flash_area_erase(ctx->flash_area, 0, ctx->flash_area->fa_size);
153 if (err) {
154 return err;
155 }
156
157 return 0;
158}

190int flash_img_prepare(struct flash_img_context *flash)
191{
192 int swap_type;
193 int err;
194
195 swap_type = mcuboot_swap_type();
196 switch (swap_type) {
197 case BOOT_SWAP_TYPE_REVERT:
198 LOG_WRN("'revert' swap type detected, it is not safe to continue");
199 return -EBUSY;
200 default:
201 LOG_INF("swap type: %s", swap_type_str(swap_type));
202 break;
203 }
204
205 err = flash_img_init(flash);
206 if (err) {
207 LOG_ERR("failed to init: %d", err);
208 return err;
209 }
210
211 err = flash_img_erase_if_needed(flash);
212 if (err) {
213 LOG_ERR("failed to erase: %d", err);
214 return err;
215 }
216
217 return 0;
218}
